INCLUDE_DIRECTORIES(
	${CMAKE_SOURCE_DIR}
	${CMAKE_SOURCE_DIR}/scribus
)

	xpsexplugin.h
)

SET(EXPORTXPS_PLUGIN_SOURCES
	xpsexplugin.cpp
)

	LIBRARY
	DESTINATION ${PLUGINDIR}
	PERMISSIONS ${PLUGIN_PERMISSIONS}
)
